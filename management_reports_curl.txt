# Admin Management Reports API Requests

# 1. Unpaid Bookings Report 
# Lists recent bookings with pending payment.
curl -X GET http://127.0.0.1:8000/api/admin/unpaid-booking-report \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Accept: application/json"

# 2. User Growth Report
# Shows new user registrations per month for the last 12 months.
curl -X GET http://127.0.0.1:8000/api/admin/user-growth-report \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Accept: application/json"
